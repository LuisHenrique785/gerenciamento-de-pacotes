// --- LÓGICA DE CONTEÚDO INFINITO E FOCADO NO ESPAÇO --- const infoDiv = document.getElementById('info'); const btn = document.getElementById('warpBtn'); // Lista de sementes para garantir que o assunto seja sempre espaço const termosEspaciais = [ "Buraco negro", "Galáxia", "Via Láctea", "Marte (planeta)", "Júpiter (planeta)", "Estrela de nêutrons", "Nebulosa", "Supernova", "Exoplaneta", "Sistema Solar", "Telescópio Espacial James Webb", "Buraco de minhoca", "Big Bang", "Constelação", "Andrômeda (galáxia)", "Quasar", "Pulsar", "Viagem espacial", "Lua", "Cometa" ]; async function buscarFatoEspacial() { infoDiv.innerHTML = 'Calculando rota para nova galáxia...'; speed = 15; try { // Sorteia um termo da lista para garantir o tema const termoAleatorio = termosEspaciais[Math.floor(Math.random() * termosEspaciais.length)]; // Busca o resumo desse termo na Wikipedia em português const url = `https://pt.wikipedia.org/w/api.php?action=query&format=json&origin=*&prop=extracts&exintro&explaintext&titles=${encodeURIComponent(termoAleatorio)}`; const response = await fetch(url); const data = await response.json(); const pages = data.query.pages; const pageId = Object.keys(pages)[0]; const content = pages[pageId].extract; if (content) { // Pega as primeiras 3 frases para não ficar um texto gigante const resumoCurto = content.split('. ').slice(0, 3).join('. ') + '.'; infoDiv.innerHTML = `${termoAleatorio}:
${resumoCurto}`; } else { buscarFatoEspacial(); // Tenta outro se der erro } } catch (e) { infoDiv.innerText = "Sinal fraco... Tente novamente a conexão intergaláctica."; } finally { setTimeout(() => speed = 0.5, 1000); } } btn.addEventListener('click', buscarFatoEspacial);
